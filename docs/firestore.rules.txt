rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {

    function signedIn() { return request.auth != null; }

    function bootstrapSuperadminEmail() {
      return signedIn() && request.auth.token.email in [
        "marceyyesi@gmail.com",
        "marceyyesituc@gmail.com"
      ];
    }

    function userDoc() {
      return get(/databases/$(database)/documents/usuarios/$(request.auth.uid));
    }

    function activeValue() {
      return signedIn() ? userDoc().data.activo : false;
    }

    function activeUser() {
      return signedIn() && (
        activeValue() == true ||
        activeValue() in ["true", "True", "TRUE", "si", "Si", "SI", "sí", "Sí", "SÍ", "activo", "Activo", "ACTIVO", "1"]
      );
    }

    function role() {
      return activeUser() ? (userDoc().data.rol) : "none";
    }

    function isEditor() {
      return bootstrapSuperadminEmail() || role() in [
        "editor", "Editor", "EDITOR",
        "admin", "Admin", "ADMIN",
        "administrador", "Administrador", "ADMINISTRADOR",
        "superadmin", "Superadmin", "SUPERADMIN"
      ];
    }

    function isAdmin() {
      return bootstrapSuperadminEmail() || role() in [
        "admin", "Admin", "ADMIN",
        "administrador", "Administrador", "ADMINISTRADOR",
        "superadmin", "Superadmin", "SUPERADMIN"
      ];
    }

    function isSuperadmin() {
      return bootstrapSuperadminEmail() || role() in ["superadmin", "Superadmin", "SUPERADMIN"];
    }

    function selfProfileFields() {
      return [
        "nombre", "nombreCompleto", "email",
        "telefono", "telefonoPerfil",
        "responsabilidad", "privilegio",
        "congregacionPerfil", "congregacion",
        "perfilBosquejos", "bosquejosPerfil",
        "discursosTiene", "discursosQuierePreparar",
        "observacionesPerfil", "perfilDiscursante",
        "perfilActualizadoEn", "updatedAt"
      ];
    }

    function selfProfileFieldsOnly() {
      return request.resource.data.diff(resource.data).affectedKeys().hasOnly(selfProfileFields());
    }

    function selfCreateProfileFieldsOnly() {
      return request.resource.data.keys().hasOnly(selfProfileFields());
    }

    match /usuarios/{uid} {
      allow read: if signedIn() && (request.auth.uid == uid || isAdmin());
      allow create: if isAdmin() || (signedIn() && request.auth.uid == uid && selfCreateProfileFieldsOnly());
      allow update: if isAdmin() || (signedIn() && request.auth.uid == uid && selfProfileFieldsOnly());
      allow delete: if isSuperadmin();
    }

    // Build 69: todo usuario autenticado puede LEER Resultados/documentos.
    // La escritura sigue limitada a editor/admin para que el usuario común no modifique asignaciones.
    match /personas/{id} {
      allow read: if signedIn();
      allow write: if isEditor();
    }

    match /asignaciones/{id} {
      allow read: if signedIn();
      allow write: if isEditor();
    }

    match /asignaciones_mensuales/{id} {
      allow read: if signedIn() || request.auth == null;
      allow write: if isEditor();
    }

    match /visitas/{id} {
      allow read: if signedIn();
      allow write: if isEditor();
    }

    match /conferenciantesLocales/{id} {
      allow read: if signedIn();
      allow write: if isEditor();
    }

    match /suplenciasDiscursantes/{id} {
      allow read: if signedIn();
      allow write: if isEditor();
    }

    match /hospitalidad_config/{id} {
      allow read: if signedIn();
      allow write: if isEditor();
    }

    match /hospitalidad_excepciones/{id} {
      allow read: if signedIn();
      allow write: if isEditor();
    }

    match /salientes/{id} {
      allow read: if signedIn() || request.auth == null;
      allow write: if isEditor();
    }

    match /discursos/{id} {
      allow read: if signedIn();
      allow write: if isAdmin();
    }

    match /canciones/{id} {
      allow read: if signedIn();
      allow write: if isAdmin();
    }
  }
}
